socket = context.socket(zmq.REQ)
socket.connect("tcp://[::1]:5555")  # 这里使用IPv6地址，根据你的设置修改

# 控制算法
def control_algorithm():
    # 在这里编写你的控制算法，根据当前状态计算控制命令
    # 这里只是一个示例，可能需要根据你的具体场景进行修改
    command = [0.1, 0.2]  # 假设控制命令为两个自由度的力或位移

# 主循环
